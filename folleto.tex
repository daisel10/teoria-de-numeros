\panel{LightBg}{%
  \phdr{Navy}{1 · Principio del Palomar y Ramsey}

  {\scriptsize La base de todo: si hay m\'as objetos que cajones,
  alg\'un caj\'on \\textbf{debe} compartirse.}

  \begin{tcolorbox}[key]
    {\scriptsize\centering
    $n{+}1$ objetos en $n$ cajones $\;\Rightarrow\;$
    alg\'un caj\'on tiene $\geq 2$ objetos.}
  \end{tcolorbox}

  {\scriptsize La \textbf{Teor\'ia de Ramsey} generaliza esta idea:
  en estructuras lo suficientemente grandes,
  \emph{siempre} emerge cierto orden.}

  \begin{tcolorbox}[def,title={N\'umero de Ramsey $R(s,t)$}]
    {\scriptsize M\'inimo $n$ tal que toda 2-coloraci\'on de $K_n$
    contiene un $K_s$ \textcolor{blue!70!black}{\bfseries azul}
    o un $K_t$ \textcolor{red!70!black}{\bfseries rojo}.}
  \end{tcolorbox}

  \begin{tcolorbox}[key]
    {\scriptsize $R(3,3)=6$ \quad $R(3,4)=9$ \quad $R(4,4)=18$}
  \end{tcolorbox}

  {\tiny\color{Navy!80}\itshape \textbf{Historia:} Frank P. Ramsey dedujo este principio en 1930 estudiando l\'ogica formal, legando toda una rama matem\'atica antes de su tr\'agica muerte a los 26 a\~nos.\par}

  \vfill
  \phdr{Navy}{2 · El Problema de la Amistad en $K_6$}

  {\scriptsize ?`Puede existir un grupo de 6 personas donde nadie
  tenga 3 amigos ni 3 extra\~nos en com\'un? \textbf{No.}}

  \begin{itemize}
    \item Colorea cada arista de $K_6$: \textcolor{red!70!black}{\bfseries rojo} = amigos,
      \textcolor{blue!70!black}{\bfseries azul} = extra\~nos.
    \item Cualquier v\'ertice tiene 5 aristas: por palomar,
      $\geq 3$ son del mismo color.
    \item Si esas 3 aristas conectan $u_1,u_2,u_3$: si alguna
      $u_iu_j$ es del mismo color $\Rightarrow$ tri\'angulo monocrom\'atico.
  \end{itemize}

  \begin{center}
  \begin{tikzpicture}[scale=0.60]
    \foreach \i/\ang in {0/90,1/30,2/-30,3/-90,4/-150,5/150}
      \coordinate (v\i) at (\ang:1.2cm);

    }

%%── PANEL 5: El Salto a los Enteros ─────────────────────────────────────────
\panel{LightBg}{%
  \phdr{Emerald}{3 · Coloraci\'on de Enteros}

  {\scriptsize ?`Qu\'e pasa si ``pintamos'' todos los n\'umeros naturales
  con un n\'umero finito de colores?}

  \begin{tcolorbox}[def,title={$r$-coloraci\'on de $\mathbb{N}$}]
    {\scriptsize Una funci\'on $f:\mathbb{N}\to\{1,\ldots,r\}$
    asigna a cada entero uno de $r$ colores.\\[2pt]
    Un conjunto $A$ es \textbf{monocrom\'atico} si todos sus
    elementos tienen el mismo color.}
  \end{tcolorbox}

  \begin{center}
  \begin{tikzpicture}[scale=0.68]
    \foreach \n/\c in {1/red,2/blue,3/Emerald,4/red,5/blue,6/Emerald,7/red,8/blue}{
      \filldraw[\c!60,draw=\c!80!black]
        ({(\n-1)*0.80cm},0) circle (0.26cm);
      \node[text=white,font=\tiny\bfseries] at ({(\n-1)*0.80cm},0) {\n};
    }
    \node[font=\small] at ({8*0.80cm},0) {$\cdots$};
  \end{tikzpicture}
  \end{center}

  {\scriptsize\textbf{Pregunta:} ?`Toda coloraci\'on de $\mathbb{N}$
  contiene siempre ternas aritm\'eticas monocrom\'aticas?}

  \vfill
  \phdr{Navy}{4 · Teorema de Schur (1916)}

  \begin{tcolorbox}[thm,title={Teorema de Schur}]
    {\scriptsize Para todo $r\geq 1$ existe $S(r)$ tal que:\\[2pt]
    toda $r$-coloraci\'on de $\{1,\ldots,n\}$ con $n\geq S(r)$
    contiene $x,y,z$ \textbf{monocrom\'aticos} con
    \[x + y = z.\]}
  \end{tcolorbox}

  {\scriptsize No importa c\'omo colorees: siempre aparece una
  terna del mismo color que suma.}

  \begin{tcolorbox}[key]
    {\scriptsize\centering
    \begin{tabular}{lrrrrr}
    $r$ & 1 & 2 & 3 & 4 & 5 \\
    $S(r)$ & 2 & 5 & 14 & 45 & 161
    \end{tabular}}
  \end{tcolorbox}

  \begin{tcolorbox}[ex,title={Aplicaci\'on notable}]
    {\scriptsize Para todo $k$ existe $n_0$ tal que para todo
    primo $p>n_0$, la ecuaci\'on\\[2pt]
    $x^k+y^k\equiv z^k\pmod{p}$\\[2pt]
    tiene soluci\'on no trivial.}
  \end{tcolorbox}
}%
%%── PANEL 6: Estructuras Lineales y Densidad ─────────────────────────────────
\panel{LightBg}{%
  \phdr{Navy}{5 · Teorema de Van der Waerden (1927)}

  \begin{tcolorbox}[def,title={Progresi\'on Aritm\'etica (PA)}]
    {\scriptsize $\{a,\;a+d,\;\ldots,\;a+(k{-}1)d\}$
    \quad tama\~no $k$, raz\'on $d$.}
  \end{tcolorbox}

  \begin{tcolorbox}[thm,title={Teorema de Van der Waerden}]
    {\scriptsize Para todo $r,k$ existe $W(r,k)$ tal que toda
    $r$-coloraci\'on de $\{1,\ldots,n\}$ con $n\geq W(r,k)$
    contiene una PA monocrom\'atica de longitud $k$.\\[2pt]
    \textbf{En $\mathbb{N}$:} toda coloraci\'on finita de los
    enteros positivos contiene PA arbitrariamente largas.}
  \end{tcolorbox}

  \begin{center}
  \begin{tikzpicture}[scale=0.68]
    \foreach \n/\c in {1/red,2/blue,3/blue,4/red,5/red,6/blue,7/blue,8/red,9/blue}{
      \filldraw[\c!60,draw=\c!80!black]
        ({(\n-1)*0.78cm},0) circle (0.26cm);
      \node[text=white,font=\tiny\bfseries] at ({(\n-1)*0.78cm},0) {\n};
    }
    \draw[blue!80!black,thick,rounded corners=2pt]

    \node[font=\tiny,blue!80!black] at ({4.5*0.78},-0.52)
      {$\{3,6,9\}$ PA azul · $d=3$};
  \end{tikzpicture}
  \end{center}

  \begin{tcolorbox}[key]
    {\scriptsize $W(2,3)=9$ \quad $W(2,4)=35$ \quad
    Cota: $W(r,k)\leq 2^{2^{r^{2^{2^{k+9}}}}}$}
  \end{tcolorbox}

  \vfill
  \phdr{Emerald}{6 · Densidad Natural}

  {\scriptsize ?`Qu\'e pasa si \textbf{no} coloreamos,
  sino que solo elegimos un conjunto ``grande''?}

  \begin{tcolorbox}[def,title={Densidad natural de $A$}]
    {\scriptsize
    $\displaystyle d(A)=\limsup_{n\to\infty}
    \frac{|A\cap\{1,\ldots,n\}|}{n}$\\[3pt]
    \begin{tabular}{ll}
    $d(2\mathbb{N})=\tfrac{1}{2}$ & $d(\{n^2\})=0$ \\
    $d(\mathbb{P})=0$ & $d(\mathbb{N})=1$
    \end{tabular}}
  \end{tcolorbox}

  \vfill
  \phdr{Amethyst}{7 · Teorema de Szemer\'edi (1975)}

  \begin{tcolorbox}[pthm,title={Teorema de Szemer\'edi}]
    {\scriptsize Todo $A\subseteq\mathbb{N}$ con $d(A)>0$
    contiene PA de \textbf{cualquier} longitud finita.}
  \end{tcolorbox}

  {\tiny\color{Amethyst!90!black}\itshape \textbf{Historia:} Paul Erd\H{o}s ofreci\'o \$1,000 USD por esta demostraci\'on. Es una obra maestra de la combinatoria que le vali\'o a Szemer\'edi el Premio Abel.\par}

  \begin{itemize}
    \item {\scriptsize \textbf{Generaliza VdW:} en una $r$-coloraci\'on,
      alguna clase tiene $d\geq 1/r>0$.}
    \item {\scriptsize \textbf{L\'imite natural:} ?`qu\'e pasa si $d(A)=0$?
      $\to$ Panel siguiente.}
  \end{itemize}
