% --- Diapositiva 1: Contexto e idea central ---
\begin{frame}{Teorema de Szemerédi: La Pregunta}

    \begin{columns}[T]

        \begin{column}{0.58\textwidth}
            \begin{block}{Conjetura de Erdős y Turán (1936)}
                Todo subconjunto de los naturales con
                \textbf{densidad positiva} contiene progresiones
                aritméticas de cualquier longitud.
            \end{block}

            \vspace{0.4cm}

            Demostrada por \textbf{Endre Szemerédi} en \textbf{1975},
            logrando el \textbf{Premio Abel}.

            \vspace{0.4cm}

            \begin{exampleblock}{Cambio de paradigma}
                \begin{tabular}{ll}
                    \textbf{Van der Waerden:} & coloración $\to$ PA\\[3pt]
                    \textbf{Szemerédi:} & densidad $\to$ PA
                \end{tabular}\\[4pt]
                Si un conjunto es lo suficientemente \textit{``grueso''},
                ya contiene progresiones.
            \end{exampleblock}
        \end{column}

        \begin{column}{0.38\textwidth}
            \begin{block}{Conexiones}
                \begin{itemize}
                    \item Combinatoria
                    \item Teoría de números
                    \item Teoría ergódica\\
                          {\footnotesize (Furstenberg 1977)}
                    \item Ciencias de la\\
                          computación (TCS)
                \end{itemize}
            \end{block}
        \end{column}

    \end{columns}

\end{frame}


% --- Diapositiva 2: Enunciado formal ---
\begin{frame}{Teorema de Szemerédi: Enunciado Formal}

    \begin{alertblock}{Teorema (Szemerédi, 1975)}
        Sea $A \subseteq \mathbb{N}$ con densidad natural positiva:
        \[
            d(A) = \limsup_{N\to\infty} \frac{|A\cap\{1,\ldots,N\}|}{N} > 0.
        \]
        Entonces $A$ contiene progresiones aritméticas de
        \textbf{cualquier longitud finita} $k \geq 1$.
    \end{alertblock}

    \pause
    \vspace{0.3cm}

    \begin{block}{¿Por qué es más fuerte que Van der Waerden?}
        \begin{itemize}
            \item \textbf{VdW} garantiza PA en \textit{algún} color
                  de una $r$-coloración completa.
            \item \textbf{Szemerédi} habla de \textit{cualquier}
                  subconjunto con densidad positiva, sin requerir
                  una partición de todos los enteros.
        \end{itemize}
    \end{block}

    \pause
    \vspace{0.15cm}
    \begin{center}
        \textit{Szemerédi implica VdW: al colorear con $r$ colores,\\
        algún color tiene densidad $\geq 1/r > 0$.}
    \end{center}

\end{frame}


% --- Diapositiva 3: Lema de regularidad y conclusión ---
\begin{frame}{El Lema de Regularidad de Szemerédi}

    Herramienta central de la demostración, hoy fundamental en
    combinatoria:

    \begin{enumerate}
        \item \textbf{Partición en bloques uniformes:} Cualquier
              grafo grande puede dividirse en bloques de tamaño similar.
        \item \textbf{Comportamiento cuasi-aleatorio:} La mayoría
              de los pares de bloques son \textit{regulares}.
        \item \textbf{Estructura forzada:} La regularidad hace
              inevitable la aparición de progresiones aritméticas.
    \end{enumerate}

    \pause
    \vspace{0.35cm}

    \begin{block}{Conclusión}
        \textbf{Si eliges una fracción constante de los enteros,
        sin importar cuán caótica sea tu elección, terminarás
        creando progresiones aritméticas perfectas.}\\[4pt]
        \centering\textit{Pura necesidad matemática.}
    \end{block}

\end{frame}
